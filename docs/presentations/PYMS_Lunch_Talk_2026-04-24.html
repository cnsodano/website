# Imports
library(httr2) # To make the API request
library(glue)  # For concatenating string parts of request

# Constants
BASE_URL <- "https://api.crossref.org"
API_VERSION <- "v1"

# Specific to this request
endpoint <- "works"
identifier <- "10.1016/s0140-6736(97)11096-0"

# https://api.crossref.org/v1/works/10.1016/s0140-6736(97)11096-0
combined_request_url = glue("{BASE_URL}/{API_VERSION}/{endpoint}/{identifier}")

req <- request(combined_request_url) # Build a "request" object
response <- req |>
    req_perform() |> # Makes API request, returns response
    resp_body_json() # Extracts JSON data from response

display_json(response) # Custom function for tutorial, uses `listviewer` package
# Imports
import requests  # To make the API request

# Constants
BASE_URL = "https://api.crossref.org"
API_VERSION = "v1"

# Specific to this request
endpoint = "works"
identifier = "10.1016/s0140-6736(97)11096-0"

# https://api.crossref.org/v1/works/10.1016/s0140-6736(97)11096-0
combined_request_url = f"{BASE_URL}/{API_VERSION}/{endpoint}/{identifier}"

req = requests.get(combined_request_url)  # Makes API request, returns response
response = req.json()  # Extracts JSON data from response

display_json(response)  # Custom function for tutorial, uses `jsonedit` javascript package
